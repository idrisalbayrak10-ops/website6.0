#A CRITICAL · SECURITY
JWT-токен пользователя логируется в console и хранится в localStorage
assets/js/google-auth.js · строки 16–17
▾
После Google OAuth вход — полный JWT ID token выводится в console.log(). Любой разработчик или расширение браузера с доступом к консоли может перехватить токен. Дополнительно весь токен сохраняется в localStorage под ключом user.token — уязвимость к XSS-атакам (достаточно одного document.cookie или уязвимого скрипта).
Баг — утечка секрета
function handleCredentialResponse(response) {
  console.log("Encoded JWT ID token: " + response.credential);
  const payload = JSON.parse(atob(response.credential.split('.')[1]));
  console.log("User Info:", payload);

  localStorage.setItem('user', JSON.stringify({
    name: payload.name,
    email: payload.email,
    picture: payload.picture,
    token: response.credential  // хранить JWT в localStorage опасно!
  }));
}
Исправление
function handleCredentialResponse(response) {
  // НЕ логировать токен в консоль!
  const payload = JSON.parse(atob(response.credential.split('.')[1]));

  // Хранить только публичные данные, не сам токен
  localStorage.setItem('user', JSON.stringify({
    name: payload.name,
    email: payload.email,
    picture: payload.picture
    // token — отправить на сервер через httpOnly cookie, не хранить в localStorage
  }));
}
⛔ Удалить console.log перед деплоем в продакшн
#B CRITICAL · SECURITY
Cloudflare Worker раскрывает внутренние ошибки пользователю
cloudflare-worker/src/index.js · строка 15
▾
При ошибке воркер возвращает error.message в теле HTTP-ответа. Это может раскрыть внутренние пути, имена переменных окружения, стек вызовов и другую чувствительную информацию злоумышленнику.
Баг
return json(
  { error: "Internal Server Error", details: error.message },
  500,
  buildCorsHeaders(request, env)
);
Исправление
// Логировать детали только в console (видно только вам в Cloudflare Dashboard)
console.error("Worker error:", error);

return json(
  { error: "Internal Server Error" },  // никаких details!
  500,
  buildCorsHeaders(request, env)
);

#F MAJOR · PERFORMANCE
favicon.svg весит 548 КБ — браузеры загружают его на каждой странице
favicon.svg (корень проекта)
▾
Файл favicon.svg весит 548 КБ. Нормальный favicon — это 1–5 КБ. Вероятно, это полноформатное изображение, переименованное в SVG. Manifest.json ссылается на него как на иконку приложения. Это означает 548 КБ лишних данных при каждом открытии страницы плюс некорректная PWA-иконка.
Текущий размер
favicon.svg → 548 КБ  (должно быть < 5 КБ)
Решение
# 1. Создать настоящий SVG-фавикон из логотипа (упрощённый путь < 5 КБ)
# 2. Для manifest.json использовать PNG-иконки:
{
  "icons": [
    { "src": "/assets/icons/icon-192.png", "sizes": "192x192", "type": "image/png" },
    { "src": "/assets/icons/icon-512.png", "sizes": "512x512", "type": "image/png" }
  ]
}
⛔ 548 КБ favicon = -0.5 балла PageSpeed на каждой странице
